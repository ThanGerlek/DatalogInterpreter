#ifndef PROJECT1_T2K_MAIN_CPP
#define PROJECT1_T2K_MAIN_CPP

#include <fstream>
#include <iostream>
// #include "Scanner.h"
// #include "Parser.h"
#include "Relation.h"

int main()
{

    Relation studentRelation("students", Scheme({"ID", "Name"}));

    std::vector<std::string> studentValues[] = {
        {"'1'", "'A'"},
        {"'3'", "'F'"}};

    for (auto &value : studentValues)
        studentRelation.addTuple(Tuple(value));

    Relation courseRelation("courses", Scheme({"ID","Name"}));

    std::vector<std::string> courseValues[] = {
        {"'1'", "'a'"}, //
        {"'2'", "'b'"},
        {"'3'", "'F'"}};

    for (auto &value : courseValues)
        courseRelation.addTuple(Tuple(value));

    std::cout << "union:" << std::endl;
    Relation result = studentRelation.unionWith(courseRelation);
    std::cout << result.toString() << std::endl;
}
